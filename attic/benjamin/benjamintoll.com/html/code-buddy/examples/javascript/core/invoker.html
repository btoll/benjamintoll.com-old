The invoker() method can be used to load up an array of functions that are invoked at once, as in when initializing an application on page load.


var invoker: function (vElem) {

  /*invoke the functions, argument can either be an array of strings or a single string*/
  /*each elem will be a namespaced string that's split by it's period, i.e. "events.setMenuHandle"*/
  /*NOTE this example uses the BLUEBOY global symbol*/
  var aNamespace = [];
  if (vElem.constructor === Array) {
    var iElemLength = vElem.length;
    for (var i = 0; i < iElemLength; i++) {
      aNamespace = vElem[i].split(".");
      BLUEBOY[aNamespace[0]][aNamespace[1]](); //invoke the function;
    }
  } else { //it's just a string;
    aNamespace = vElem.split(".");
    BLUEBOY[aNamespace[0]][aNamespace[1]](); //invoke the function;
  }

};

The invocation below would invoke the specified functions in the BLUEBOY.utility, BLUEBOY.events, and BLUEBOY.global namespaces.


invoker([
  "utility.targetBlank",
  "events.setAccordion",
  "events.setMenuHandle",
  "events.setRegex",
  "events.setSearchEngine",
  "events.setSorting",
  "global.createCategoriesList"
]);
