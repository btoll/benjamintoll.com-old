Brief explanation: Calling the native object method Array.prototype.push and using the obj object as its context allows the object to be treated as if it were an array, such as modifying the length property.


var obj = {
  length: 0,
  push: function (v) {
    Array.prototype.push.call(this, v);
    return this; //allows methods to be chained;
  }
};
obj.push([1, 2, 3]).push(500);
console.log(obj.length, obj[0], obj[1]); //results = 2 [1, 2, 3] 500;
